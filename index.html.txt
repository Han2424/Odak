  for(var i=0;i<SUBJECTS.length;i++){ if(SUBJECTS[i].id===id) return SUBJECTS[i]; }
  return SUBJECTS[SUBJECTS.length-1];
}

function uid(){ return Date.now().toString(36)+Math.random().toString(36).slice(2,8); }

function pad2(n){ return n<10 ? "0"+n : ""+n; }

function dateKey(d){
  return d.getFullYear()+"-"+pad2(d.getMonth()+1)+"-"+pad2(d.getDate());
}
function todayKey(){ return dateKey(new Date()); }

function fmtClock(totalSeconds){
  totalSeconds = Math.max(0, Math.round(totalSeconds));
  var h = Math.floor(totalSeconds/3600);
  var m = Math.floor((totalSeconds%3600)/60);
  var s = totalSeconds%60;
  if(h>0) return h+":"+pad2(m)+":"+pad2(s);
  return pad2(m)+":"+pad2(s);
}

function fmtDurationShort(minutesFloat){
  var totalMin = Math.round(minutesFloat);
  var h = Math.floor(totalMin/60);
  var m = totalMin%60;
  if(h>0) return h+" sa " + (m>0? m+" dk":"");
  return m+" dk";
}

function fmtHourLabel(h){
  return pad2(h)+":00";
}

var Store = {
  sessions: [],
  load: function(){
    try{
      var raw = localStorage.getItem(STORAGE_KEY);
      this.sessions = raw ? JSON.parse(raw) : [];
    }catch(e){ this.sessions = []; }
  },
  save: function(){
    try{ localStorage.setItem(STORAGE_KEY, JSON.stringify(this.sessions)); }catch(e){}
  },
  add: function(session){
    this.sessions.push(session);
    this.save();
  },
  remove: function(id){
    this.sessions = this.sessions.filter(function(s){ return s.id!==id; });
    this.save();
  },
  byDate: function(key){
    return this.sessions.filter(function(s){ return s.dateKey===key; });
  },
  allSortedDesc: function(){
    return this.sessions.slice().sort(function(a,b){ return b.startTime - a.startTime; });
  }
};

/* ============================================================
   SAYAÇ MOTORU — zaman damgası tabanlı (arka planda doğru çalışır)
============================================================ */
var Timer = {
  state: null, // { mode, periodMinutes, subjectId, topic, startTime, accumPausedMs, pausedAt, running }
  tickHandle: null,

  load: function(){
    try{
      var raw = localStorage.getItem(TIMER_KEY);
      this.state = raw ? JSON.parse(raw) : null;
    }catch(e){ this.state = null; }
  },
  persist: function(){
    try{
      if(this.state) localStorage.setItem(TIMER_KEY, JSON.stringify(this.state));
      else localStorage.removeItem(TIMER_KEY);
    }catch(e){}
  },
  start: function(mode, periodMinutes, subjectId, topic){
    this.state = {
      mode: mode, // 'period' | 'free'
      periodMinutes: periodMinutes || null,
      subjectId: subjectId,
      topic: topic,
      startTime: Date.now(),
      accumPausedMs: 0,
      pausedAt: null,
      running: true,
      autoFinished: false
    };
    this.persist();
    this.startTick();
  },
  pause: function(){
    if(!this.state || !this.state.running) return;
    this.state.running = false;
    this.state.pausedAt = Date.now();
    this.persist();
  },
  resume: function(){
    if(!this.state || this.state.running) return;
    var pausedDuration = Date.now() - this.state.pausedAt;
    this.state.accumPausedMs += pausedDuration;
    this.state.pausedAt = null;
    this.state.running = true;
    this.persist();
  },
  elapsedMs: function(){
    if(!this.state) return 0;
    var end = this.state.running ? Date.now() : (this.state.pausedAt || Date.now());
    return Math.max(0, end - this.state.startTime - this.state.accumPausedMs);
  },
  remainingMs: function(){
    if(!this.state || this.state.mode!=="period") return 0;
    return (this.state.periodMinutes*60*1000) - this.elapsedMs();
  },
  clear: function(){
    this.state = null;
    this.persist();
    this.stopTick();
  },
  startTick: function(){
    this.stopTick();
    var self = this;
    this.tickHandle = setInterval(function(){ self.onTick && self.onTick(); }, 1000);
  },
  stopTick: function(){
    if(this.tickHandle){ clearInterval(this.tickHandle); this.tickHandle = null; }
  }
};

function renderCalendar(){
  var panel = document.getElementById("panel-calendar");
  var year = UI.calYear, month = UI.calMonth;
  var firstDay = new Date(year, month, 1);
  var startOffset = (firstDay.getDay()+6)%7; // Pazartesi=0
  var daysInMonth = new Date(year, month+1, 0).getDate();
  var todayK = todayKey();

  // o ayda günlere göre toplam süre haritası
  var dayTotals = {};
  Store.sessions.forEach(function(s){
    var d = s.dateKey;
    dayTotals[d] = (dayTotals[d]||0) + s.durationMinutes;
  });
  var maxMin = 1;
  Object.keys(dayTotals).forEach(function(k){ if(dayTotals[k]>maxMin) maxMin = dayTotals[k]; });

  var cells = "";
  for(var i=0;i<startOffset;i++){ cells += '<div class="cal-day empty"></div>'; }
  for(var day=1; day<=daysInMonth; day++){
    var d = new Date(year, month, day);
    var k = dateKey(d);
    var total = dayTotals[k] || 0;
    var hasData = total>0;
    var intensity = hasData ? Math.max(0.18, Math.min(0.85, total/maxMin)) : 0;
    var classes = "cal-day" + (k===todayK ? " today":"") + (hasData ? " has-data":"");
    cells += '<div class="'+classes+'" data-key="'+k+'">'+
      '<div class="intensity-bg" style="opacity:'+intensity+'"></div>'+
      '<span class="cal-day-num">'+day+'</span>'+
      (hasData ? '<span class="dot" style="position:relative; z-index:1;"></span>':'')+
    '</div>';
  }

  panel.innerHTML =
    '<div class="cal-header">'+
      '<div class="cal-month">'+MONTHS_TR[month]+' '+year+'</div>'+
      '<div class="cal-nav">'+
        '<button id="calPrev"><svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.4"><path d="M15 18l-6-6 6-6"/></svg></button>'+
        '<button id="calNext"><svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.4"><path d="M9 18l6-6-6-6"/></svg></button>'+
      '</div>'+
    '</div>'+
    '<div class="card">'+
      '<div class="cal-weekdays">'+WEEKDAYS_TR.map(function(w){return '<div>'+w+'</div>';}).join("")+'</div>'+
      '<div class="cal-grid">'+cells+'</div>'+
    '</div>'+
    '<div class="cal-legend"><div class="legend-box" style="background:var(--accent); opacity:.2"></div>az &nbsp; <div class="legend-box" style="background:var(--accent); opacity:.85"></div>çok</div>';

  document.getElementById("calPrev").addEventListener("click", function(){
    UI.calMonth--; if(UI.calMonth<0){ UI.calMonth=11; UI.calYear--; }
    renderCalendar();
  });
  document.getElementById("calNext").addEventListener("click", function(){
    UI.calMonth++; if(UI.calMonth>11){ UI.calMonth=0; UI.calYear++; }
    renderCalendar();
  });
  panel.querySelectorAll(".cal-day.has-data").forEach(function(cell){
    cell.addEventListener("click", function(){
      openDayDetail(cell.getAttribute("data-key"));
    });
  });
}

function openDayDetail(key){
  var sessions = Store.byDate(key).sort(function(a,b){return a.startTime-b.startTime;});
  if(sessions.length===0) return;
  var overlay = document.getElementById("dayModalOverlay");
  var totalMin = sessions.reduce(function(a,s){return a+s.durationMinutes;},0);
  var avgEff = sessions.reduce(function(a,s){return a+s.efficiency;},0)/sessions.length;
  var dateObj = new Date(key+"T00:00:00");
  var label = DAYS_LONG_TR[(dateObj.getDay()+6)%7]+", "+dateObj.getDate()+" "+MONTHS_TR[dateObj.getMonth()];

  var rows = sessions.map(function(s){
    var subj = subjectById(s.subjectId);
    return sessionRowHTML(s, subj, true);
  }).join("");

  overlay.innerHTML =
    '<div class="modal-sheet">'+
      '<div class="modal-handle"></div>'+
      '<div class="modal-title">'+label+'</div>'+
      '<div class="day-detail-summary">'+
        '<div class="dd-stat"><b>'+fmtDurationShort(totalMin)+'</b><span>TOPLAM</span></div>'+
        '<div class="dd-stat"><b>'+sessions.length+'</b><span>SEANS</span></div>'+
        '<div class="dd-stat"><b>'+avgEff.toFixed(1)+'</b><span>ORT. VERİM</span></div>'+
      '</div>'+
      '<div>'+rows+'</div>'+
    '</div>';
  overlay.classList.add("show");
  wireSessionDeleteButtons(overlay, function(){
    openDayDetail(key);
    renderCalendar(); renderLog(); renderStats();
    if(Store.byDate(key).length===0) overlay.classList.remove("show");
  });
}

function renderLog(){
  var panel = document.getElementById("panel-log");
  var sessions = Store.sessions;
  if(sessions.length===0){
    panel.innerHTML = emptyStateHTML("Henüz bir kayıt yok", "İlk çalışma seansını tamamladığında günlük burada birikmeye başlayacak.");
    return;
  }

  var byDay = {};
  sessions.forEach(function(s){
    if(!byDay[s.dateKey]) byDay[s.dateKey] = [];
    byDay[s.dateKey].push(s);
  });
  var dateKeys = Object.keys(byDay).sort().reverse();

  panel.innerHTML = '<div class="section-title">Güne göre çalışma & verim</div>' +
    dateKeys.map(function(k){
      var daySessions = byDay[k].sort(function(a,b){return a.startTime-b.startTime;});
      var totalMin = daySessions.reduce(function(a,s){return a+s.durationMinutes;},0);
      var avgEff = daySessions.reduce(function(a,s){return a+s.efficiency;},0)/daySessions.length;
      var dObj = new Date(k+"T00:00:00");
      var label = (k===todayKey()) ? "Bugün" : DAYS_LONG_TR[(dObj.getDay()+6)%7]+", "+dObj.getDate()+" "+MONTHS_TR[dObj.getMonth()];
      var isOpen = !!UI.openLogDates[k];
      var effColor = effColorFor(avgEff);

      var rows = daySessions.map(function(s){
        return sessionRowHTML(s, subjectById(s.subjectId), true);
      }).join("");

      return '<div class="log-day'+(isOpen?' open':'')+'" data-day="'+k+'">'+
        '<div class="log-day-head" data-toggle="'+k+'">'+
          '<div class="log-day-left">'+
            '<div class="log-day-date">'+label+'</div>'+
            '<div class="log-day-sub">'+daySessions.length+' seans</div>'+
          '</div>'+
          '<div class="log-day-right">'+
            '<span class="log-time-badge">'+fmtDurationShort(totalMin)+'</span>'+
            '<span class="eff-badge" style="background:'+effColor.bg+'; color:'+effColor.fg+'">'+avgEff.toFixed(1)+'</span>'+
            '<svg class="chev" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.2"><path d="M6 9l6 6 6-6"/></svg>'+
          '</div>'+
        '</div>'+
        '<div class="log-day-body" id="logbody-'+k+'"><div class="log-day-body-inner">'+rows+'</div></div>'+
      '</div>';
    }).join("");

  panel.querySelectorAll("[data-toggle]").forEach(function(head){
    head.addEventListener("click", function(){
      var k = head.getAttribute("data-toggle");
      UI.openLogDates[k] = !UI.openLogDates[k];
      var dayEl = panel.querySelector('[data-day="'+k+'"]');
      var body = document.getElementById("logbody-"+k);
      dayEl.classList.toggle("open", UI.openLogDates[k]);
      body.style.maxHeight = UI.openLogDates[k] ? body.scrollHeight+"px" : "0px";
    });
  });
  wireSessionDeleteButtons(panel, function(){ renderLog(); renderCalendar(); renderStats(); });
}

function effColorFor(eff){
  if(eff>=7.5) return { bg:"rgba(51,217,166,0.18)", fg:"#33D9A6" };
  if(eff>=5) return { bg:"rgba(245,166,35,0.18)", fg:"#F5A623" };
  return { bg:"rgba(241,85,108,0.18)", fg:"#F1556C" };
}

function refreshRunningDisplay(){
  var st = Timer.state;
  if(!st) return;
  var pauseBtn = document.getElementById("pauseBtn");
  var statePill = document.getElementById("statePill");
  var ringWrap = document.getElementById("ringWrap");
  if(!pauseBtn) return;

  if(st.running){
    pauseBtn.innerHTML = '<svg width="17" height="17" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><rect x="6" y="4" width="4" height="16" rx="1"/><rect x="14" y="4" width="4" height="16" rx="1"/></svg>Duraklat';
    statePill.innerHTML = '<span class="pulse-dot"></span>Çalışıyor';
    statePill.classList.remove("paused");
    ringWrap.classList.add("running");
  } else {
    pauseBtn.innerHTML = '<svg width="17" height="17" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M6 4l14 8-14 8V4z"/></svg>Devam Et';
    statePill.textContent = "Duraklatıldı";
    statePill.classList.add("paused");
    ringWrap.classList.remove("running");
  }

  var timeDisplay = document.getElementById("timeDisplay");
  var ringProgress = document.getElementById("ringProgress");
  var circumference = 2*Math.PI*118;

  if(st.mode==="period"){
    var remainMs = Timer.remainingMs();
    if(remainMs<=0 && !st.autoFinished){
      st.autoFinished = true;
      Timer.persist();
      timeDisplay.textContent = "00:00";
      ringProgress.style.strokeDashoffset = "0";
      handlePeriodComplete();
      return;
    }
    timeDisplay.textContent = fmtClock(remainMs/1000);
    var frac = Math.max(0, Math.min(1, remainMs/(st.periodMinutes*60*1000)));
    ringProgress.style.strokeDashoffset = (circumference*(1-frac)).toFixed(1);
  } else {
    var elapsed = Timer.elapsedMs();
    timeDisplay.textContent = fmtClock(elapsed/1000);
    // serbest modda halka 60 dakikalık döngüyle dolup boşalır (görsel nabız)
    var loopFrac = (elapsed % (60*60*1000)) / (60*60*1000);
    ringProgress.style.strokeDashoffset = (circumference*(1-loopFrac)).toFixed(1);
  }
}

function handlePeriodComplete(){
  try{
    var ctx = new (window.AudioContext||window.webkitAudioContext)();
    [880,1175].forEach(function(freq, i){
      var o = ctx.createOscillator(), g = ctx.createGain();
      o.connect(g); g.connect(ctx.destination);
      o.frequency.value = freq;
      o.type = "sine";
      g.gain.setValueAtTime(0.0001, ctx.currentTime + i*0.18);
      g.gain.exponentialRampToValueAtTime(0.18, ctx.currentTime + i*0.18 + 0.02);
      g.gain.exponentialRampToValueAtTime(0.0001, ctx.currentTime + i*0.18 + 0.4);
      o.start(ctx.currentTime + i*0.18);
      o.stop(ctx.currentTime + i*0.18 + 0.45);
    });
  }catch(e){}
  if(navigator.vibrate){ try{ navigator.vibrate([120,60,120]); }catch(e){} }
  showToast("Süre tamamlandı! Nasıl geçti?");
  openFinishFlow(true);
}

function wireGlobalTimerEvents(){
  Timer.onTick = function(){
    if(UI.activeTab==="timer") refreshRunningDisplay();
  };
  document.addEventListener("visibilitychange", function(){
    if(document.visibilityState==="visible" && Timer.state && UI.activeTab==="timer"){
      refreshRunningDisplay();
    }
  });
}

/* ---------- Bitirme akışı + Verimlilik modalı ---------- */
function openFinishFlow(autoTriggered){
  if(!Timer.state) return;
  var elapsedMs = Timer.elapsedMs();
  if(Timer.state.mode==="period" && autoTriggered){
    elapsedMs = Timer.state.periodMinutes*60*1000;
  }
  if(elapsedMs < 1000){
    // çok kısa süre, kaydetmeden direkt iptal
    Timer.clear();
    renderTimerTab();
    return;
  }
  UI.pendingFinishElapsedMs = elapsedMs;
  showEfficiencyModal();
}

function finalizeSession(efficiency){
  var st = Timer.state;
  var minutes = UI.pendingFinishElapsedMs/60000;
  var startDate = new Date(st.startTime);
  Store.add({
    id: uid(),
    dateKey: dateKey(startDate),
    subjectId: st.subjectId,
    topic: st.topic,
    durationMinutes: Math.round(minutes*10)/10,
    mode: st.mode,
    plannedMinutes: st.periodMinutes || null,
    startTime: st.startTime,
    endTime: Date.now(),
    startHour: startDate.getHours(),
    efficiency: efficiency
  });
  Timer.clear();
  UI.setup.topic = "";
  UI.setup.subjectId = null;
  showToast("Harika iş! Seansın kaydedildi ✓");
  renderTimerTab();
  renderCalendar();
  renderLog();
  renderStats();
}

function renderStats(){
  var panel = document.getElementById("panel-stats");
  var sessions = Store.sessions;

  if(sessions.length===0){
    panel.innerHTML = emptyStateHTML("Henüz analiz edilecek veri yok", "Birkaç seans tamamladığında burada detaylı bir analiz panosu seni karşılayacak.");
    return;
  }

  var totalMin = sessions.reduce(function(a,s){return a+s.durationMinutes;},0);
  var avgEff = sessions.reduce(function(a,s){return a+s.efficiency;},0)/sessions.length;
  var totalSessions = sessions.length;
  var streak = computeStreak();

  var html = '';

  // ---- KPI kartları
  html += '<div class="kpi-grid">'+
    kpiCard(fmtDurationShort(totalMin), "Toplam çalışma", "accent")+
    kpiCard(totalSessions+"", "Toplam seans", "")+
    kpiCard(avgEff.toFixed(1)+"/10", "Ortalama verim", "accent2")+
    kpiCard(streak+" gün", "Güncel seri 🔥", "")+
  '</div>';

  // ---- Günlük çalışma grafiği
  html += dailyChartCardHTML(sessions);

  // ---- Derslere göre toplam süre
  html += subjectComparisonCardHTML(sessions, totalMin);

  // ---- Saatlere göre verim
  html += hourlyEfficiencyCardHTML(sessions);

  // ---- Derslere göre ortalama verim (mini sıralama)
  html += subjectEfficiencyCardHTML(sessions);

  // ---- Otomatik içgörüler
  html += insightsCardHTML(sessions, avgEff);

  panel.innerHTML = html;
  wireStatsEvents(sessions);
}

function kpiCard(value, label, accentClass){
  return '<div class="kpi-card"><div class="kpi-value'+(accentClass?' kpi-'+accentClass:'')+'">'+value+'</div><div class="kpi-label">'+label+'</div></div>';
}

function computeStreak(){
  var datesWithData = {};
  Store.sessions.forEach(function(s){ datesWithData[s.dateKey] = true; });
  var streak = 0;
  var cursor = new Date();
  while(true){
    var k = dateKey(cursor);
    if(datesWithData[k]){ streak++; cursor.setDate(cursor.getDate()-1); }
    else if(k===todayKey()){ cursor.setDate(cursor.getDate()-1); continue; } // bugün henüz çalışılmamış olabilir, seriyi bozma
    else break;
  }
  return streak;
}

function dailyChartCardHTML(sessions){
  var range = UI.statsRange;
  var byDay = {};
  sessions.forEach(function(s){ byDay[s.dateKey] = (byDay[s.dateKey]||0) + s.durationMinutes; });

  var days = [];
  var cursor = new Date();
  for(var i=range-1;i>=0;i--){
    var d = new Date(); d.setDate(cursor.getDate()-i);
    var k = dateKey(d);
    days.push({ key:k, label: d.getDate()+" "+MONTHS_TR[d.getMonth()].slice(0,3), minutes: byDay[k]||0, dow: d.getDay() });
  }
  var maxMin = Math.max(1, Math.max.apply(null, days.map(function(d){return d.minutes;})));

  var bars = days.map(function(d){
    var h = Math.max(3, Math.round((d.minutes/maxMin)*120));
    var showLabel = (range<=14) || (d.dow===1) || (d.minutes===maxMin && d.minutes>0);
    return '<div class="daily-bar-wrap" title="'+d.label+': '+fmtDurationShort(d.minutes)+'">'+
      '<div class="daily-bar" style="height:'+h+'px; opacity:'+(d.minutes>0?1:0.25)+'"></div>'+
      '<div class="daily-label">'+(showLabel? d.label : "")+'</div>'+
    '</div>';
  }).join("");

  return '<div class="card chart-card">'+
    '<div class="chart-card-head">'+
      '<div><div class="chart-title">Günlük çalışma süresi</div><div class="chart-sub">Son '+range+' gün</div></div>'+
      '<div class="range-switch" id="rangeSwitch">'+
        [7,14,30].map(function(r){ return '<button data-range="'+r+'" class="'+(UI.statsRange===r?'active':'')+'">'+r+'g</button>'; }).join("")+
      '</div>'+
    '</div>'+
    '<div class="daily-chart">'+bars+'</div>'+
  '</div>';
}

function subjectComparisonCardHTML(sessions, totalMin){
  var bySubject = {};
  sessions.forEach(function(s){ bySubject[s.subjectId] = (bySubject[s.subjectId]||0) + s.durationMinutes; });
  var rows = Object.keys(bySubject).map(function(id){
    return { subj: subjectById(id), minutes: bySubject[id] };
  }).sort(function(a,b){ return b.minutes-a.minutes; });
  var maxMin = Math.max.apply(null, rows.map(function(r){return r.minutes;}));

  var bars = rows.map(function(r){
    var pct = totalMin>0 ? Math.round((r.minutes/totalMin)*100) : 0;
    var widthPct = Math.max(4, Math.round((r.minutes/maxMin)*100));
    return '<div class="subject-bar-row">'+
      '<div class="sb-top"><span class="sb-name"><span class="chip-dot" style="background:'+r.subj.color+'"></span>'+r.subj.name+'</span>'+
      '<span class="sb-val">'+fmtDurationShort(r.minutes)+' · %'+pct+'</span></div>'+
      '<div class="sb-track"><div class="sb-fill" style="width:'+widthPct+'%; background:'+r.subj.color+'"></div></div>'+
    '</div>';
  }).join("");

  return '<div class="card chart-card">'+
    '<div class="chart-card-head"><div><div class="chart-title">Derslere göre dağılım</div><div class="chart-sub">Toplam süre karşılaştırması</div></div></div>'+
    bars+
  '</div>';
}

function hourlyEfficiencyCardHTML(sessions){
  var buckets = {}; // hour -> {sum, count}
  sessions.forEach(function(s){
    var h = s.startHour;
    if(!buckets[h]) buckets[h] = { sum:0, count:0 };
    buckets[h].sum += s.efficiency;
    buckets[h].count++;
  });

  var hours = [];
  for(var h=0; h<24; h++){
    var b = buckets[h];
    hours.push({ hour:h, avg: b ? b.sum/b.count : 0, has: !!b });
  }
  // sadece veri olan saat aralığını göster (boş baştan/sondan kırp), min 6 sütun
  var firstIdx = hours.findIndex(function(x){return x.has;});
  var lastIdx = -1;
  for(var i=hours.length-1;i>=0;i--){ if(hours[i].has){ lastIdx=i; break; } }
  if(firstIdx===-1){ firstIdx=8; lastIdx=22; }
  firstIdx = Math.max(0, firstIdx-1);
  lastIdx = Math.min(23, lastIdx+1);
  var visible = hours.slice(firstIdx, lastIdx+1);

  var bars = visible.map(function(x){
    var heightPct = x.has ? Math.max(8, (x.avg/10)*100) : 4;
    return '<div class="hour-bar-wrap" title="'+fmtHourLabel(x.hour)+' · '+(x.has? x.avg.toFixed(1)+'/10':'veri yok')+'">'+
      '<div class="hour-bar'+(x.has?' has-data':'')+'" style="height:'+heightPct+'%"></div>'+
      '<div class="hour-label">'+x.hour+'</div>'+
    '</div>';
  }).join("");

  var best = hours.filter(function(x){return x.has;}).sort(function(a,b){return b.avg-a.avg;})[0];

  return '<div class="card chart-card">'+
    '<div class="chart-card-head"><div><div class="chart-title">Saate göre verim</div>'+
      '<div class="chart-sub">'+(best ? "En verimli saatin: "+fmtHourLabel(best.hour) : "Henüz yeterli veri yok")+'</div></div></div>'+
    '<div class="hour-chart">'+bars+'</div>'+
  '</div>';
}

function subjectEfficiencyCardHTML(sessions){

    bySubject[s.subjectId].sum += s.efficiency;
    bySubject[s.subjectId].count++;
    bySubject[s.subjectId].minutes += s.durationMinutes;
  });
  var rows = Object.keys(bySubject).map(function(id){
    var b = bySubject[id];
    return { subj: subjectById(id), avg: b.sum/b.count, minutes:b.minutes };
  }).sort(function(a,b){ return b.avg-a.avg; });

  var maxAvg = Math.max.apply(null, rows.map(function(r){return r.avg;}));

  var bars = rows.map(function(r){
    var widthPct = Math.max(6, Math.round((r.avg/10)*100));
    return '<div class="subject-bar-row">'+
      '<div class="sb-top"><span class="sb-name"><span class="chip-dot" style="background:'+r.subj.color+'"></span>'+r.subj.name+'</span>'+
      '<span class="sb-val">'+r.avg.toFixed(1)+'/10</span></div>'+
      '<div class="sb-track"><div class="sb-fill" style="width:'+widthPct+'%; background:'+r.subj.color+'"></div></div>'+
    '</div>';
  }).join("");

  return '<div class="card chart-card">'+
    '<div class="chart-card-head"><div><div class="chart-title">Derslere göre ortalama verim</div><div class="chart-sub">Hangi derste daha odaklısın?</div></div></div>'+
    bars+
  '</div>';
}

  var topSubj = subjectById(topSubjId);
  insights.push({
    color: topSubj.color, bg: hexToSoft(topSubj.color),
    icon: ICONS.log,
    text: 'En çok zaman ayırdığın ders <b>'+topSubj.name+'</b> — toplam '+fmtDurationShort(bySubjMin[topSubjId])+'.'
  });

  // en yüksek verimli ders
  var bySubjEff = {};
  sessions.forEach(function(s){
    if(!bySubjEff[s.subjectId]) bySubjEff[s.subjectId] = {sum:0,count:0};
    bySubjEff[s.subjectId].sum += s.efficiency; bySubjEff[s.subjectId].count++;
  });
  var bestEffId = Object.keys(bySubjEff).sort(function(a,b){
    return (bySubjEff[b].sum/bySubjEff[b].count) - (bySubjEff[a].sum/bySubjEff[a].count);
  })[0];
  var bestEffSubj = subjectById(bestEffId);
  var bestEffVal = bySubjEff[bestEffId].sum/bySubjEff[bestEffId].count;
  insights.push({

function hexToSoft(hex){
  var r = parseInt(hex.slice(1,3),16), g = parseInt(hex.slice(3,5),16), b = parseInt(hex.slice(5,7),16);
  return 'rgba('+r+','+g+','+b+',0.15)';
}
